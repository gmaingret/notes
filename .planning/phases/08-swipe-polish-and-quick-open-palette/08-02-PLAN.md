    **Search result sections** (useMemo, only when query.length >= 2):
    ```typescript
    const docResults = useMemo(() => {
      if (query.length < 2 || !allDocs) return [];
      return allDocs
        .filter(d => d.title.toLowerCase().includes(query.toLowerCase()))
        .slice(0, 3);
    }, [allDocs, query]);

    const bulletResults = (searchResults ?? []).slice(0, 3);
    const bookmarkResults = useMemo(() => {
      if (query.length < 2 || !bookmarks) return [];
      return bookmarks
        .filter(b => b.content.toLowerCase().includes(query.toLowerCase()))
        .slice(0, 3);
    }, [bookmarks, query]);
    ```

    **Flat result list for keyboard nav** (useMemo):
    ```typescript
    const flatResults: PaletteResult[] = useMemo(() => {
      if (query.length < 2) {
        return recentDocs.map(doc => ({ type: 'document' as const, doc }));
      }
      return [
        ...docResults.map(doc => ({ type: 'document' as const, doc })),
        ...bulletResults.map(result => ({ type: 'bullet' as const, result })),
        ...bookmarkResults.map(bookmark => ({ type: 'bookmark' as const, bookmark })),
      ];
    }, [query, recentDocs, docResults, bulletResults, bookmarkResults]);
    ```

      return () => window.removeEventListener('keydown', onKey);
    }, [onClose]);

    **Keyboard navigation handler** (on the input element's onKeyDown):
    ```typescript
    function handleInputKeyDown(e: React.KeyboardEvent<HTMLInputElement>) {
      if (e.key === 'ArrowDown') {
        e.preventDefault();
        setSelectedIndex(i => Math.min(i + 1, flatResults.length - 1));
      } else if (e.key === 'ArrowUp') {
        e.preventDefault();
        setSelectedIndex(i => Math.max(i - 1, 0));
      } else if (e.key === 'Enter') {
        e.preventDefault();
        if (flatResults[selectedIndex]) openResult(flatResults[selectedIndex]);
      }
    }
    ```

    **openResult function:**
    ```typescript
    function openResult(item: PaletteResult) {
      if (item.type === 'document') {
        navigate(`/doc/${item.doc.id}`);
      } else if (item.type === 'bullet') {
        navigate(`/doc/${item.result.documentId}#bullet/${item.result.bulletId}`);
      } else if (item.type === 'bookmark') {
        navigate(`/doc/${item.bookmark.documentId}#bullet/${item.bookmark.bulletId}`);
      }
      onClose();
    }
    ```
