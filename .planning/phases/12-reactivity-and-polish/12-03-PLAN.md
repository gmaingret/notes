2. Create `SearchViewModel.kt`:
   ```kotlin
   @HiltViewModel
   class SearchViewModel @Inject constructor(
       private val searchBulletsUseCase: SearchBulletsUseCase
   ) : ViewModel() {
       private val _uiState = MutableStateFlow<SearchUiState>(SearchUiState.Idle)
       val uiState: StateFlow<SearchUiState> = _uiState.asStateFlow()

       private val queryFlow = MutableSharedFlow<String>(replay = 0, extraBufferCapacity = 64)

       init {
           viewModelScope.launch {
               queryFlow.debounce(300).collect { query ->
                   if (query.isBlank()) {
                       _uiState.value = SearchUiState.Empty
                   } else {
                       _uiState.value = SearchUiState.Loading
                       searchBulletsUseCase(query).fold(
                           onSuccess = { results ->
                               if (results.isEmpty()) _uiState.value = SearchUiState.Empty
                               else _uiState.value = SearchUiState.Success(results.groupBy { it.documentTitle })
                           },
                           onFailure = { _uiState.value = SearchUiState.Error(it.message ?: "Search failed") }
                       )
                   }
               }
           }
       }

       fun onQueryChange(query: String) { viewModelScope.launch { queryFlow.emit(query) } }
       fun reset() { _uiState.value = SearchUiState.Idle }
   }
   ```

4. Create `BookmarksViewModel.kt`:
   ```kotlin
   @HiltViewModel
   class BookmarksViewModel @Inject constructor(
       private val getBookmarksUseCase: GetBookmarksUseCase
   ) : ViewModel() {
       private val _uiState = MutableStateFlow<BookmarksUiState>(BookmarksUiState.Loading)
       val uiState: StateFlow<BookmarksUiState> = _uiState.asStateFlow()

       init { loadBookmarks() }

       fun loadBookmarks() {
           viewModelScope.launch {
               _uiState.value = BookmarksUiState.Loading
               getBookmarksUseCase().fold(
                   onSuccess = { bookmarks ->
                       _uiState.value = if (bookmarks.isEmpty()) BookmarksUiState.Empty
                                        else BookmarksUiState.Success(bookmarks)
                   },
                   onFailure = { _uiState.value = BookmarksUiState.Error(it.message ?: "Failed to load bookmarks") }
               )
           }
       }
   }
   ```
