            DisplayState.NOT_CONFIGURED -> WidgetUiState.NotConfigured

    **1. BulletTreeViewModel — add triggerWidgetRefreshIfNeeded():**

    Add a private method that checks if the current document matches the widget's pinned document, and if so, fetches fresh data and writes it to the widget cache:

    ```kotlin
    private fun triggerWidgetRefreshIfNeeded() {
        val docId = currentDocumentId ?: return
        viewModelScope.launch {
            try {
                val store = WidgetStateStore.getInstance(getApplication())
                val pinnedDocId = store.getFirstDocumentId()
                if (pinnedDocId == null || pinnedDocId != docId) return@launch

                // Fetch fresh data using WidgetEntryPoint (same as worker)
                val entryPoint = EntryPointAccessors.fromApplication(
                    getApplication(),
                    WidgetEntryPoint::class.java
                )
                val widget = NotesWidget()
                val uiState = widget.fetchWidgetData(entryPoint, docId)

                // Write result to cache
                when (uiState) {
                    is WidgetUiState.Content -> {
                        store.saveBullets(uiState.bullets)
                        store.saveDisplayState(DisplayState.CONTENT)
                    }
                    is WidgetUiState.Empty -> {
                        store.saveBullets(emptyList())
                        store.saveDisplayState(DisplayState.EMPTY)
                    }
                    is WidgetUiState.SessionExpired -> store.saveDisplayState(DisplayState.SESSION_EXPIRED)
                    is WidgetUiState.DocumentNotFound -> store.saveDisplayState(DisplayState.DOCUMENT_NOT_FOUND)
                    is WidgetUiState.Error -> store.saveDisplayState(DisplayState.ERROR)
                    else -> { /* Loading, NotConfigured — ignore */ }
                }

                // Trigger widget re-render from cache
                widget.updateAll(getApplication())
            } catch (_: Exception) {
                // Silent failure — widget will catch up on next WorkManager sync
            }
        }
    }
    ```

    Add required imports: `WidgetStateStore`, `WidgetEntryPoint`, `DisplayState`, `NotesWidget`, `WidgetUiState`, `dagger.hilt.android.EntryPointAccessors`.
