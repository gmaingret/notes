```typescript
  onDelete: (bulletId: string, documentId: string) => void;
}

export function useSwipeGesture(
  bullet: { id: string; documentId: string; isComplete: boolean },
  actions: SwipeActions,
) {
  // ... all swipe state and handlers

  return {
    rowRef,
    swipeX,
    isSwiping,
    exitDirection,
    showUndoToast,
    setShowUndoToast,
    swipeBackground,
    iconScale,

    handleRowPointerDown,
    handleRowPointerMove,
    handleRowPointerUp,
    // Context menu touch handlers
    handleCtxTouchStart,
    handleCtxTouchMove,
    handleCtxTouchEnd,

```

The `handleTransitionEnd` function takes the React.TransitionEvent and performs the same logic currently inline in the `onTransitionEnd` prop (lines 368-381): checks `propertyName === 'transform'`, reads `pendingActionRef`, calls the appropriate action callback, then resets state.

The `isCtxTouchActive` function replaces the `ctxTouchStart.current` null-check used in the `onContextMenu` handler (line 311) — BulletNode uses it to decide whether to show the context menu on right-click.
